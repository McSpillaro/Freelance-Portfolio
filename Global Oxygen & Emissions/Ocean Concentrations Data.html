# Creating the raw dataframe of all unorganized data
df = pd.read_csv('Global Oxygen18.csv')

# Removing '**' values in dataframe
df = df[df['Salinity']!= '**']
df = df[df['Year']!= '**']
df = df[df['Month']!= '**']
df = df[df['pTemperature']!= '**']
df = df[df['Depth']!='**']

# Converting all 'Years', 'Salinity', and 'Months' to type:str to type:int
df['Salinity'] = pd.to_numeric(df['Salinity'])
df['Year'] = pd.to_numeric(df['Year'])
df['Month'] = pd.to_numeric(df['Month'])
df['pTemperature'] = pd.to_numeric(df['pTemperature'])
# df['Depth'] = pd.to_numeric(df['Depth'])

# Renaming columns
df = df.rename(columns={'Salinity':'Salinity (ppt)'})
